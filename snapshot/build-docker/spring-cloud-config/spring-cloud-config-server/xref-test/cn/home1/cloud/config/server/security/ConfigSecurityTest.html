43      @Test
44      public void testVerifyParentPasswordToken() {
45          final String application = "application";
46          final String parentApplication = "parent-application";
47          final String parentPassword = "parent-application_pass";
48          final String token = this.configSecurity.encryptParentPassword(application, parentApplication, parentPassword);
49  
50          log.info("application: {}, parent-application: {}, parent-application_pass: {}, token: {}", //
51              application, parentApplication, parentPassword, token);
52  
53          assertTrue(this.configSecurity.verifyParentPassword(application, parentApplication, token, parentPassword));
54          assertFalse(this.configSecurity.verifyParentPassword(application, "", token, parentPassword));
55          assertFalse(this.configSecurity.verifyParentPassword(application, null, token, parentPassword));
56          assertFalse(this.configSecurity.verifyParentPassword("", parentApplication, token, parentPassword));
57          assertFalse(this.configSecurity.verifyParentPassword(null, parentApplication, token, parentPassword));
58          assertFalse(this.configSecurity.verifyParentPassword(application + "x", parentApplication, token, parentPassword));
59          assertFalse(this.configSecurity.verifyParentPassword(application, parentApplication + "x", token, parentPassword));
60          assertFalse(this.configSecurity.verifyParentPassword(application, parentApplication, token, parentPassword + "x"));
61      }

63      @Test
64      public void testVerifyNonTokenParentPassword() {
65          final String application = "application";
66          final String parentApplication = "parent-application";
67          final String parentPassword = "parent-application_pass";
68  
69          final String encryptedParentPassword = "{cipher}" + this.encryptor.encrypt(parentPassword);
70  
71          assertTrue(this.configSecurity.verifyParentPassword(application, parentApplication, parentPassword, parentPassword));
72          assertTrue(this.configSecurity.verifyParentPassword(application, parentApplication, parentPassword, encryptedParentPassword));
73          assertTrue(this.configSecurity.verifyParentPassword(application, parentApplication, encryptedParentPassword, parentPassword));
74  
75          // plain text parent password does not contain application info, can not verify by application.
76          assertTrue(this.configSecurity.verifyParentPassword(application, "", parentPassword, parentPassword));
77          assertTrue(this.configSecurity.verifyParentPassword(application, null, parentPassword, parentPassword));
78  
79          assertFalse(this.configSecurity.verifyParentPassword("", parentPassword, parentPassword, parentPassword));
80          assertFalse(this.configSecurity.verifyParentPassword(null, parentPassword, parentPassword, parentPassword));
81          assertFalse(this.configSecurity.verifyParentPassword(application, parentApplication, parentPassword + "x", parentPassword));
82          assertFalse(this.configSecurity.verifyParentPassword(application, parentApplication, parentPassword, parentPassword + "x"));
83      }

85      @Test
86      public void testVerifyEmptyParentPassword() {
87          final String application = "application";
88          final String parentApplication = "parent-application";
89  
90          assertTrue(this.configSecurity.verifyParentPassword(application, parentApplication, "", null));
91          assertTrue(this.configSecurity.verifyParentPassword(application, parentApplication, null, ""));
92          assertTrue(this.configSecurity.verifyParentPassword(application, parentApplication, null, null));
93          assertFalse(this.configSecurity.verifyParentPassword(application, "", null, null));
94          assertFalse(this.configSecurity.verifyParentPassword(application, null, null, null));
95          assertFalse(this.configSecurity.verifyParentPassword("", parentApplication, null, null));
96          assertFalse(this.configSecurity.verifyParentPassword(null, parentApplication, null, null));
97      }

99      /**
100      * see: {@link org.springframework.cloud.config.server.config.EncryptionAutoConfiguration}
101      */
102     TextEncryptorLocator textEncryptorLocator() {
103         final KeyStore keyStore = new KeyStore();
104         keyStore.setAlias("key_alias");
105         keyStore.setSecret("key_pass");
106         keyStore.setLocation(new ClassPathResource("default-keystore.jks"));
107         keyStore.setPassword("store_pass");
108 
109         final RsaProperties rsaProperties = new RsaProperties();
110 
111         KeyStoreTextEncryptorLocator locator = new KeyStoreTextEncryptorLocator(
112             new KeyStoreKeyFactory(keyStore.getLocation(), keyStore.getPassword().toCharArray()),
113             keyStore.getSecret(), keyStore.getAlias());
114         RsaAlgorithm algorithm = rsaProperties.getAlgorithm();
115         locator.setRsaAlgorithm(algorithm);
116         locator.setSalt(rsaProperties.getSalt());
117         locator.setStrong(rsaProperties.isStrong());
118         return locator;
119     }
