77    private <T> HttpEntity<T> structHttpEntityWithAuthHeaders(final String name, final String password) {
78      HttpHeaders firstAppAuthHeaders = new HttpHeaders();
79      String firstAppNamePwdPair = name + ":" + password;
80      String base64EncodeAuthStr = new String(Base64.getEncoder().encode(firstAppNamePwdPair.getBytes(UTF8)), UTF8);
81      firstAppAuthHeaders.add("Authorization", "Basic " + base64EncodeAuthStr);
82      return new HttpEntity<>(firstAppAuthHeaders);
83    }

85    @Test

87      String requestUrl = host + "/" + firstAppName + "/development" + DOT_ENV;
88      log.info(requestUrl);

90      log.info(result.getStatusCode().toString());
91      Assert.assertTrue(result.getStatusCode() == HttpStatus.UNAUTHORIZED);
