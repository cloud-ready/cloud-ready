13  /**
14   * use url instead of filename.
15   * see: {@link org.springframework.cloud.config.monitor.PropertyPathEndpoint#guessServiceName(String)}
16   * see: {@link org.springframework.cloud.config.monitor.GitlabPropertyPathNotificationExtractor}
17   * see: {@link org.springframework.cloud.config.monitor.EnvironmentMonitorAutoConfiguration}
18   * Created by zhanghaolun on 16/11/2.

20  @Order(Ordered.LOWEST_PRECEDENCE - 100)
21  public class GitlabRepositoryPathNotificationExtractor extends AbstraceNotificationExtractor {
22  
23    @Override
24    public PropertyPathNotification extract( //
25        final MultiValueMap<String, String> headers, //
26        final Map<String, Object> payload //
27    ) {
28      final PropertyPathNotification result;
29      if ("Push Hook".equals(headers.getFirst("X-Gitlab-Event"))) {
30        if (payload.get("commits") instanceof Collection) {
31          final Set<String> paths = new HashSet<>();
32  
33          @SuppressWarnings("unchecked") final Collection<Map<String, Object>> commits = (Collection<Map<String, Object>>) payload.get("commits");
34  
35          for (final Map<String, Object> commit : commits) {
36            addAllPaths(paths, commit, "added");
37            addAllPaths(paths, commit, "removed");
38            addAllPaths(paths, commit, "modified");
39          }
40  
41          result = !paths.isEmpty() ? new PropertyPathNotification(paths.toArray(new String[0])) : null;
42        } else {
43          result = null;
44        }
45      } else {
46        result = null;
47      }
48      return result;
49    }
50  }
