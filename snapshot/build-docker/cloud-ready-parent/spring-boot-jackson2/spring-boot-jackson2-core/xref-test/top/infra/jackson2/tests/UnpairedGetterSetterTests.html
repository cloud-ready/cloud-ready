47          this.rawObjectMapper = new ObjectMapper();
48      }
49  
50      @Test
51      @SneakyThrows
52      public void testMissingSetter() {
53          final Input input1 = new Input();
54          input1.setName("input1");
55          input1.setValue("input1Value");
56          final Output.OutputDetail outputDetail1 = new Output.OutputDetail();
57          outputDetail1.setField1("outputDetail1Field1");
58          outputDetail1.setField2("outputDetail1Field2");
59          final Output output1 = new Output();
60          output1.setInputName(input1.getName());
61          output1.setDetails(Lists.newArrayList(outputDetail1));
62  
63          final Field field1 = new Field();
64          field1.setName(FIELD1);
65          field1.setValue("value1");
66          field1.setStatus(OK);
67  
68          final Field field2 = new Field();
69          field2.setName(FIELD2);
70          field2.setValue("value2");
71          field2.setStatus(DELETED);
72  
73          final ResultFields<Output> resultFields = new ResultFields<>();
74          resultFields.setOutput(output1);
75          resultFields.put(FIELD1, field1);
76          resultFields.put(FIELD2, field2);
77  
78          final SubResultFieldsMap data = new SubResultFieldsMap(input1);
79          data.put(SOURCE1, resultFields);
80  
81          log.info("data customObjectMapper: {}", this.customObjectMapper.writeValueAsString(data));
82          log.info("data rawObjectMapper: {}", this.rawObjectMapper.writeValueAsString(data));
83          log.info("data springObjectMapper: {}", this.springObjectMapper.writeValueAsString(data));
84  
85          assertEquals(this.springObjectMapper.writeValueAsString(data), this.customObjectMapper.writeValueAsString(data));
86          assertEquals(this.customObjectMapper.writeValueAsString(data), this.rawObjectMapper.writeValueAsString(data));
87  
88          final SubResultFieldsMap readed = this.springObjectMapper.readValue(
89              this.springObjectMapper.writeValueAsString(data), SubResultFieldsMap.class);
90          assertEquals(data, readed);
91      }
