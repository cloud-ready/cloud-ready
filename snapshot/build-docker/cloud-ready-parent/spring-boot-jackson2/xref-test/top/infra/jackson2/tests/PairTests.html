43      @Test
44      @SneakyThrows
45      public void testPair() {
46          final Pair<PairKey, PairValue> pair = newImmutablePair();
47  
48          log.info("pair customObjectMapper: {}", this.customObjectMapper.writeValueAsString(pair));
49          log.info("pair rawObjectMapper: {}", this.rawObjectMapper.writeValueAsString(pair));
50          log.info("pair springObjectMapper: {}", this.springObjectMapper.writeValueAsString(pair));
51          assertEquals(this.springObjectMapper.writeValueAsString(pair), this.customObjectMapper.writeValueAsString(pair));
52          assertNotEquals(this.springObjectMapper.writeValueAsString(pair), this.rawObjectMapper.writeValueAsString(pair));
53  
54          final String serialized = this.springObjectMapper.writeValueAsString(pair);
55          final Pair<PairKey, PairValue> deserialized = this.springObjectMapper.readValue(
56              serialized,
57              new TypeReference<Pair<PairKey, PairValue>>() {
58              }
59          );
60          log.info("deserialized springObjectMapper: {}", deserialized);
61          assertEquals(pair, deserialized);
62  
63          final Pair<PairKey, PairValue> pairNullKey = new ImmutablePair<>(null, pair.getValue());
64          log.info("pairNullKey springObjectMapper: {}", this.springObjectMapper.writeValueAsString(pairNullKey));
65          final Pair<PairKey, PairValue> pairNullValue = new ImmutablePair<>(pair.getKey(), null);
66          log.info("pairNullValue springObjectMapper: {}", this.springObjectMapper.writeValueAsString(pairNullValue));
67          final Pair<PairKey, PairValue> pairNull = new ImmutablePair<>(null, null);
68          log.info("pairNull springObjectMapper: {}", this.springObjectMapper.writeValueAsString(pairNull));
69      }

71      @Test
72      @SneakyThrows
73      public void testNestedPair() {
74          final Pair<PairKey, PairValue> pair = newImmutablePair();
75          final Pair<PairKey, Pair<PairKey, PairValue>> nestedPair = new ImmutablePair<>(pair.getKey(), pair);
76  
77          log.info("nestedPair customObjectMapper: {}", this.customObjectMapper.writeValueAsString(nestedPair));
78          log.info("nestedPair rawObjectMapper: {}", this.rawObjectMapper.writeValueAsString(nestedPair));
79          log.info("nestedPair springObjectMapper: {}", this.springObjectMapper.writeValueAsString(nestedPair));
80          assertEquals(this.springObjectMapper.writeValueAsString(pair), this.customObjectMapper.writeValueAsString(pair));
81          assertNotEquals(this.springObjectMapper.writeValueAsString(pair), this.rawObjectMapper.writeValueAsString(pair));
82      }

85      @SneakyThrows
86      public void testPairInBean() {
87          final Pair<PairKey, PairValue> pair = newImmutablePair();
88          final Pair<PairKey, Pair<PairKey, PairValue>> nestedPair = new ImmutablePair<>(pair.getKey(), pair);
89          final Entity entity = new Entity();
90          entity.setPair(nestedPair);
91  
92          log.info("entity customObjectMapper: {}", this.customObjectMapper.writeValueAsString(entity));
93          log.info("entity rawObjectMapper: {}", this.rawObjectMapper.writeValueAsString(entity));
94          log.info("entity springObjectMapper: {}", this.springObjectMapper.writeValueAsString(entity));
95          assertEquals(this.springObjectMapper.writeValueAsString(pair), this.customObjectMapper.writeValueAsString(pair));
96          assertNotEquals(this.springObjectMapper.writeValueAsString(pair), this.rawObjectMapper.writeValueAsString(pair));
97      }
