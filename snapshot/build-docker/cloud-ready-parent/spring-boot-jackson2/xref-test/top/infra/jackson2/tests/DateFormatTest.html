22  @RunWith(SpringRunner.class)
23  @SpringBootTest(classes = Jackson2TestApplication.class)
24  @Slf4j
25  public class DateFormatTest {
26  
27      static {
28          System.setProperty("spring.jackson.date-format", DefaultJackson2Customizer.PATTERN_JAVA_ISO8601);
29          System.setProperty("spring.jackson.serialization.write_dates_as_timestamps", "false");
30          System.setProperty("spring.jackson.time-zone", "UTC");
31      }
32  
33      @Autowired
34      private Jackson2Properties jackson2Properties;
35  
36      @Autowired
37      private ObjectMapper objectMapper;
38  
39      @Test
40      public void testDateFormat() throws JsonProcessingException {
41          final Date now = new Date();
42          final String nowSerialized = this.objectMapper.writeValueAsString(now).replaceAll("\"", "");
43          log.info("nowSerialized testDateFormat: {}", nowSerialized);
44  
45          final SimpleDateFormat simpleDateFormat = new SimpleDateFormat(DefaultJackson2Customizer.PATTERN_JAVA_ISO8601);
46          simpleDateFormat.setTimeZone(this.jackson2Properties.getTimeZone());
47          assertEquals(simpleDateFormat.format(now), nowSerialized);
48      }
