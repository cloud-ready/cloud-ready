19      @Test
20      public void testDateFormat() throws JsonProcessingException {
21          final Jackson2Properties jackson2Properties = new Jackson2Properties();
22          jackson2Properties.setDateFormat(Jackson2MapperCustomizer.PATTERN_JAVA_ISO8601);
23  
24          final JodaTimeJackson2Customizer customizer = new JodaTimeJackson2Customizer();
25          final ObjectMapper objectMapper = new ObjectMapper();
26          objectMapper.configure(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS, false);
27          customizer.customize(jackson2Properties, objectMapper);
28  
29          final DateTime now = DateTime.now(DateTimeZone.UTC);
30          final String nowSerialized = objectMapper.writeValueAsString(now).replaceAll("\"", "");
31          log.info("nowSerialized testDateFormat: {}", nowSerialized);
32          assertTrue(nowSerialized.startsWith("{"));
33          assertTrue(nowSerialized.endsWith("}"));
34      }
35  
36      @Test
37      public void testJodaDateTimeFormat() throws JsonProcessingException {
38          final Jackson2Properties jackson2Properties = new Jackson2Properties();
39          jackson2Properties.setJodaDateTimeFormat("yyyy-MM-dd'T'HH:mm:ss.SSSZ");
40  
41          final JodaTimeJackson2Customizer customizer = new JodaTimeJackson2Customizer();
42          final ObjectMapper objectMapper = new ObjectMapper();
43          objectMapper.configure(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS, false);
44          customizer.customize(jackson2Properties, objectMapper);
45  
46          final DateTime now = DateTime.now(DateTimeZone.UTC);
47          final String nowSerialized = objectMapper.writeValueAsString(now).replaceAll("\"", "");
48  
49          log.info("nowSerialized testJodaDateTimeFormat: {}", nowSerialized);
50          assertTrue(nowSerialized.endsWith("+0000"));
51          final DateTime nowParsed = customizer
52              .getJacksonJodaDateFormat(jackson2Properties)
53              .withTimeZone(DateTimeZone.UTC.toTimeZone())
54              .rawFormatter()
55              .parseDateTime(nowSerialized);
56          assertEquals(now, nowParsed);
57      }
