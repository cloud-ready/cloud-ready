38      /**
39       * Setup a Mapper.
40       *
41       * @param properties   jackson2Properties
42       * @param objectMapper ObjectMapper or XmlMapper to customize
43       */
44      public static void customize(final Jackson2Properties properties, final ObjectMapper objectMapper) {
45          for (final Jackson2MapperCustomizer customizer : jackson2MapperCustomizers()) {
46              log.info("customize objectMapper: '{}' using: '{}'.", objectMapper, customizer);
47              customizer.customize(properties, objectMapper);
48          }
49      }

51      public static <T> Function<String, T> fromJson( //
52          final ObjectMapper objectMapper, //
53          final TypeReference<T> typeReference //
54      ) {
55          return string -> {
56              try {
57                  return objectMapper.readValue(string, typeReference);
58              } catch (final IOException wrapped) {
59                  throw new RuntimeJsonProcessingException("error read from JSON.", wrapped);
60              }
61          };
62      }

72      public static <T> Function<String, T> fromJson( //

89      static List<Jackson2MapperCustomizer> jackson2MapperCustomizers() {
90          final List<Jackson2MapperCustomizerFactory> factories = jackson2MapperCustomizerFactories();
91  
92          return factories.stream()
93              .map(factory -> factory.getObject().orElse(null))
94              .filter(Objects::nonNull)
95              .sorted(OrderComparator.INSTANCE)
96              .collect(toList());
97      }
98  
99      static List<Jackson2MapperCustomizerFactory> jackson2MapperCustomizerFactories() {
100         final String basePackage = Jackson2Utils.class.getName().split("\\.")[0];
101 
102         final Set<Class<Jackson2MapperCustomizerFactory>> classes = FileAndClasspathUtils.scan(basePackage,
103             new FileAndClasspathUtils.AssignableFilter(Jackson2MapperCustomizerFactory.class, false, false));
104 
105         log.info("found {} jackson2MapperCustomizerFactories.", classes.size());
106 
107         return classes.stream().map(BeanUtils::instantiate).collect(toList());
108     }

110     /**
111      * to JSON.
112      *
113      * @param objectMapper objectMapper
114      * @param <T>          type
115      * @return JSON
116      */
117     public static <T> Function<T, String> toJson(final ObjectMapper objectMapper) {
118         return object -> {
119             try {
120                 if (object != null) {
121                     return objectMapper.writeValueAsString(object);
122                 } else {
123                     return "";
124                 }
125             } catch (final JsonProcessingException wrapped) {
126                 throw new RuntimeJsonProcessingException("error serialize to JSON.", wrapped);
127             }
128         };
129     }
130 
131     public static <T> String toJson(final ObjectMapper objectMapper, final T item) {
132         return toJson(objectMapper).apply(item);
133     }
