7   import java.util.concurrent.atomic.AtomicInteger;

9   import javax.servlet.http.HttpServletRequest;
10  import javax.servlet.http.HttpServletResponse;
11  
12  import top.infra.cloudready.boot.OrderedHandlerInterceptorAdapter;
13  
14  /**
15   * common log interceptor
16   */
17  @Slf4j
18  public class TestInterceptor extends OrderedHandlerInterceptorAdapter {
19  
20    private AtomicInteger count = new AtomicInteger(0);
21  
22    @Override
23    public boolean preHandle(final HttpServletRequest req, final HttpServletResponse resp, final Object o) throws Exception {
24      log.info("preHandle {}", this.count.incrementAndGet());
25      return true;
26    }
27  
28    @Override
29    public void afterCompletion(final HttpServletRequest req, final HttpServletResponse resp, final Object o, final Exception ex) throws Exception {
30      log.info("afterCompletion", this.count.incrementAndGet());
31    }
32  
33    @Override
34    public int getOrder() {
35      return Ordered.LOWEST_PRECEDENCE;
36    }
37  
38    public int getCount() {
39      return this.count.get();
40    }
41  }
