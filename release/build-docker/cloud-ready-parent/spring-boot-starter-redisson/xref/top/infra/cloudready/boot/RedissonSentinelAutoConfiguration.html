1   package top.infra.cloudready.boot;

3   import lombok.extern.slf4j.Slf4j;
4   
5   import org.redisson.Redisson;
6   import org.redisson.api.RedissonClient;
7   import org.redisson.config.Config;
8   import org.redisson.config.SentinelServersConfig;
9   import org.springframework.beans.factory.annotation.Autowired;
10  import org.springframework.boot.autoconfigure.AutoConfigureAfter;
11  import org.springframework.boot.autoconfigure.condition.ConditionalOnClass;
12  import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
13  import org.springframework.boot.autoconfigure.data.redis.RedisAutoConfiguration;
14  import org.springframework.boot.autoconfigure.data.redis.RedisProperties;
15  import org.springframework.context.annotation.Bean;
16  import org.springframework.context.annotation.Configuration;
17  import org.springframework.data.redis.core.RedisTemplate;
18  
19  import java.util.regex.Pattern;
20  
21  @AutoConfigureAfter(RedisAutoConfiguration.class)
22  @ConditionalOnClass({Redisson.class, RedisTemplate.class})
23  @ConditionalOnProperty(name = {"spring.redis.sentinel.master", "spring.redis.sentinel.nodes"})
24  @Configuration
25  @Slf4j
26  public class RedissonSentinelAutoConfiguration {
27  
28    private RedisProperties redisProperties;
29  
30    @Autowired(required = false)
31    public void setRedisProperties(final RedisProperties redisProperties) {
32      this.redisProperties = redisProperties;
33    }
34  
35    public String[] getRedisSentinelNodes() {
36      //should add schema prefix "redis://" or "rediss://" for redisson version >= 3.7.0
37      final String schema = this.redisProperties.isSsl() ? "rediss://" : "redis://";
38      final String[] redisSentinelNodes = Pattern.compile(",") //
39          .splitAsStream(this.redisProperties.getSentinel().getNodes()) //
40          .map(node -> schema + node) //
41          .toArray(String[]::new);
42      log.info("redisNodes: {}", (Object) redisSentinelNodes);
43      return redisSentinelNodes;
44    }
45  
46    private SentinelServersConfig sentinelServers(final Config config) {
47      final int redisDatabase = this.redisProperties.getDatabase();
48      final String redisPassword = this.redisProperties.getPassword();
49      final String redisSentinelMaster = this.redisProperties.getSentinel().getMaster();
50      final String[] redisSentinelNodes = this.getRedisSentinelNodes();
51  
52      SentinelServersConfig sentinelServersConfig = config.useSentinelServers();
53      sentinelServersConfig.setMasterName(redisSentinelMaster);
54      sentinelServersConfig.addSentinelAddress(redisSentinelNodes);
55      sentinelServersConfig.setDatabase(redisDatabase);
56      if (redisPassword != null) { //StringUtils.isNotBlank(redisPassword)
57        sentinelServersConfig.setPassword(redisPassword);
58      }
59  
60      return sentinelServersConfig;
61    }
62  
63    @Bean
64    public RedissonClient redissonClient() {
65      final Config config = new Config();
66      this.sentinelServers(config);
67      return Redisson.create(config);
68    }
69  }
