1   package top.infra.cloudready.boot;
2   
3   import lombok.extern.slf4j.Slf4j;
4   
5   import org.redisson.Redisson;
6   import org.redisson.api.RedissonClient;
7   import org.redisson.config.Config;
8   import org.redisson.config.SingleServerConfig;
9   import org.springframework.beans.factory.annotation.Autowired;
10  import org.springframework.boot.autoconfigure.AutoConfigureAfter;
11  import org.springframework.boot.autoconfigure.condition.ConditionalOnClass;
12  import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
13  import org.springframework.boot.autoconfigure.data.redis.RedisAutoConfiguration;
14  import org.springframework.boot.autoconfigure.data.redis.RedisProperties;
15  import org.springframework.context.annotation.Bean;
16  import org.springframework.context.annotation.Configuration;
17  import org.springframework.data.redis.core.RedisTemplate;
18  
19  @AutoConfigureAfter(RedisAutoConfiguration.class)
20  @ConditionalOnClass({Redisson.class, RedisTemplate.class})
21  @ConditionalOnProperty(name = {"spring.redis.host", "spring.redis.port"})
22  @Configuration
23  @Slf4j
24  public class RedissonStandaloneAutoConfiguration {
25  
26      private RedisProperties redisProperties;
27  
28      @Autowired(required = false)
29      public void setRedisProperties(final RedisProperties redisProperties) {
30          this.redisProperties = redisProperties;
31      }
32  
33      public String getRedisAddress() {
34          //should add schema prefix "redis://" or "rediss://" for redisson version >= 3.7.0
35          final String schema = this.redisProperties.isSsl() ? "rediss://" : "redis://";
36          final String redisAddress = schema + this.redisProperties.getHost() + ":" + this.redisProperties.getPort();
37          log.info("redisAddress: {}", redisAddress);
38          return redisAddress;
39      }
40  
41      private SingleServerConfig singleServer(final Config config) {
42          final SingleServerConfig singleServerConfig = config.useSingleServer();
43  
44          final String redisAddress = this.getRedisAddress();
45          final int redisDatabase = this.redisProperties.getDatabase();
46          final String redisPassword = this.redisProperties.getPassword();
47  
48          singleServerConfig.setAddress(redisAddress).setDatabase(redisDatabase);
49          if (redisPassword != null) { //StringUtils.isNotBlank(redisPassword)
50              singleServerConfig.setPassword(redisPassword);
51          }
52  
53          return singleServerConfig;
54      }
55  
56      @Bean
57      public RedissonClient redissonClient() {
58          final Config config = new Config();
59          this.singleServer(config);
60          return Redisson.create(config);
61      }
62  }
