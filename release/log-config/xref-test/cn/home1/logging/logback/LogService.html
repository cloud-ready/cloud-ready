18  @Service
19  public class LogService {
20  
21    public void doLog(final String loggerName, final Level level, final String message) {
22      this.doLog(loggerName, null, level, message);
23    }
24  
25    public void doLog(final String loggerName, final Marker marker, final Level level, final String message) {
26      final Logger logger = isNotBlank(loggerName) ? //
27          LoggerFactory.getLogger(loggerName) : LoggerFactory.getLogger(LogService.class);
28  
29      switch (level) {
30        case TRACE: {
31          logger.trace(marker, message);
32          break;
33        }
34        case DEBUG: {
35          logger.debug(marker, message);
36          break;
37        }
38        case INFO: {
39          logger.info(marker, message);
40          break;
41        }
42        case WARN: {
43          logger.warn(marker, message);
44          break;
45        }
46        case ERROR: {
47          logger.error(marker, message);
48          break;
49        }
50        default: {
51          throw new IllegalArgumentException("unknown log level: " + level.name());
52        }
53      }
54    }
55  
56    public String getLogProperty(final String propertyName) {
57      final LoggerContext lc = (LoggerContext) LoggerFactory.getILoggerFactory();
58      return lc.getProperty(propertyName);
59    }
60  }
